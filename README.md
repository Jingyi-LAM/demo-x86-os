# demo-x86-os

## [Install]
apt install -f nasm gcc bochs-x

## [Run]
user@pc:~$ cd demo-x86-os/mainline

user@pc:~/demo-x86-os/mainline$ sudo ./build.sh

user@pc:~/demo-x86-os/mainline$ bochs
